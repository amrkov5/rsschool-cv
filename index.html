Custom Array.map() 
                        
                            Array.prototype.map = function(foo,thisArg = 0) {
                                const arr = [];
                                let that = this;
                                let callback = foo;
                                if(thisArg) {
                                    callback = foo.bind(thisArg);
                                };
                                for (let i=0; i<that.length; i++) {
                                    if(!(i in that)) {
                                        arr.push(that[i]);
                                    } else {
                                        arr.push(callback(that[i],i,that));
                                    };
                                };
                                return arr;
                            }
